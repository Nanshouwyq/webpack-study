# 深度优先遍历

先访问根节点
对根节点的没有访问过的相邻节点挨个进行深度优先遍历

const graph ={
  0:[1,2],
  1:[2],
  2:[0,3],
  3:[3]
}
const visited = new Set()
const dfs =(n)=>{
 console.log(n)
 visited.add(n)
 graph[n].forEach(element => {
   if(!visited.has(element)){
     dfs(element)
   }
 });
}
dfs(2)

# 广度优先遍历

新建一个队列，把根节点入队
把队头出队并访问
把队头没有访问过的相邻的节点入队
重复2,3步直到队列为空

const graph ={
  0:[1,2],
  1:[2],
  2:[0,3],
  3:[3]
}
const visited = new Set()
const bfs =(n)=>{
 const q=[n];
 const visited = new Set()
 visited.add(n)
 while(q.length){
  const m=q.shift()
  console.log(m)  
  graph[m].forEach(element => {
    if(!visited.has(element)){
      q.push(element)
      visited.add(element)
    }
  });
 } 
}
bfs(2)
