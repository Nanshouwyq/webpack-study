# 排序和搜索

1.排序：把某个乱序的数组变成升序或者降序的数组
2.搜索：找出数组中的某个元素的下标
JS中的排序：数组的sort方法 JS中的搜索：数组的indexOf方法

# 冒泡排序

1.比较所有相邻的元素，如果第一个比第二个大，交换 2.一轮下来，最后一个元素是最大的 3.重复n-1轮 时间复杂度o(n^2)

Array.prototype.bubbleSort = function(){
  for(let i=0;i<this.length -1;i++){
    for(let j=0;j<this.length -1-i; j++){
      if(this[j] > this[j+1]){
        const tem =this[j]
        this[j] = this[j+1]
        this[j+1]= tem
      }
    }
  }
 
}
const arr = [5,7,2,3,4]
arr.bubbleSort()
console.log(arr)

# 选择排序

1.选择最小的放在第一位 2.选择第二小的放在第二位
3.执行n-1轮 时间复杂度O(n^2)

Array.prototype.searchSort = function() {
    for (let i = 0; i < this.length - 1; i++) {
        let indexMin = i;
        for (j = i + 1; j < this.length; j++) {
            if (this[j] < this[indexMin]) {
                indexMin = j
            }
        }
        if (i !== indexMin) {
            let temp = this[i]
            this[i] = this[indexMin]
            this[indexMin] = temp
        }

    }
}
const arr = [5, 7, 2, 3, 4]
arr.searchSort()

# 插入排序

在小型数组排序中插入排序比冒泡，选择排序性能好
1.从第二个数开始往前比
2.比它大就往后排
3.以此类推进行到最后一个数

Array.prototype.sectionSort = function() {
    for (let i = 1; i < this.length; i++) {
        let temp = this[i]
        let j = i;
        while (j > 0) {
            if (this[j - 1] > temp) {
                this[j] = this[j - 1]
            } else {
                break;
            }
            j--;
        }
        this[j] = temp
    }
}
const arr = [5, 7, 2, 3, 4]
arr.sectionSort()
console.log(arr)

# 归并排序

分：把数组劈成两半，再递归对子数组进行“分”操作，直到分成一个个单独的数
合：把两个数合并为有序数组，再对有序数组进行合并，直到全部子数组合并为一个完整数组

# 合并两个有序数组

1.新建一个空数组res 存放最终排序后的数组
2.比较两个数组的头部，最小的出队添加到res中 3.如果两个数组中都有值，重复第二步 分时间复杂度 为O(logn) 合时间复杂度为 O(n) 和起来为O(nLogn)

Array.prototype.mergeSort = function() {
    const rec = (arr) => {
        if (arr.length === 1) return arr
        let mid = Math.floor(arr.length / 2)
        let left = arr.slice(0, mid)
        let right = arr.slice(mid, arr.length)
        const orderLeft = rec(left)
        const orderRight = rec(right)
        const res = []
        while (orderLeft.length || orderRight.length) {
            if (orderLeft.length && orderRight.length) {
                res.push(orderLeft[0] < orderRight[0] ? orderLeft.shift() : orderRight.shift())
            } else if (orderLeft.length) {
                res.push(orderLeft.shift())
            } else if (orderRight.length) {
                res.push(orderRight.shift())
            }
        }
        return res
    }
    const res = rec(this)
    res.forEach((val, key) => {
        this[key] = val
    });

}
const arr = [5, 7, 2, 3, 4]
arr.mergeSort()
console.log(arr)

# 快速排序

1.随便找个基准为mid 小 进入left队列 大进入right队列
2.递归执行子队列
递归的时间复杂度为O(logN)
分区的时间复杂度为0(n)
合一起的时间复杂度为O(nlogN)

Array.prototype.quickSort = function() {
    const rec = (arr) => {
        if (arr.length === 1) return arr
        let mid = arr[0]
        const left = [];
        const right = [];
        for (let i = 1; i < arr.length; i++) {
            if (arr[i] < mid) {
                left.push(arr[i])
            } else {
                right.push(arr[i])
            }
        }
        const leftRes = left.length ? rec(left) : []
        const rightRes = right.length ? rec(right) : []
        return [...leftRes, mid, ...rightRes]
    }
    const res = rec(this)
    res.forEach((val, key) => {
        this[key] = val
    });
}
const arr = [5, 7, 2, 3, 4]
arr.quickSort()
console.log(arr)

# 顺序搜索

1.遍历数组
2.找到目标值相等的元素，返回它的下标
3.没有找到返回-1

Array.prototype.sequentialSearch = function(n) {
    for (let i = 0; i < this.length; i++) {
        if (this[i] === n) {
            return i
        }
    }
    return -1
}
const arr = [5, 7, 2, 3, 4]
const n = arr.sequentialSearch(7)
console.log(n)

# 二分搜索

有序数组使用二分搜索
1.从数组的中间元素开始，如果中间元素正好是目标值，然后直接中间值下标
2.目标值在大于或者小于目标值 在一半的数组已经进行搜索

Array.prototype.sequentialSearch = function(n) {
    let low = 0;
    let high = this.length - 1;
    while (low <= high) {
        let mid = Math.floor((low + high) / 2)
        if (n > this[mid]) {
            low = mid + 1
        } else if (n < this[mid]) {
            high = mid - 1
        } else {
            return mid
        }
    }
    return -1
}
const arr = [1, 2, 3, 4, 5, 6, 7]
const n = arr.sequentialSearch(4)
console.log(n)
